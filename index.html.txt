  <script>
    /*
      Change this GAME_ID for future events.
      Example: "kade-bingo-board-2-v1"
      This forces everyone to receive a fresh shuffled card.
    */
    const GAME_ID = "kade-bingo-board-1-v1";

    const phrases = [
      "Goes to fill up water",
      "Monster",
      "Flying",
      "Hit a T20",
      "In the mixer",
      "Straight Red",
      "Giulangi Buffon",
      "Over 120 on the dart board",
      "No auctions"
    ];

    /*
      Row or column only.
      No diagonals.
    */
    const winningLines = [
      [0, 1, 2],
      [3, 4, 5],
      [6, 7, 8],
      [0, 3, 6],
      [1, 4, 7],
      [2, 5, 8]
    ];

    const card = document.getElementById("bingoCard");
    const message = document.getElementById("message");

    let cardPhrases = [];
    let markedSquares = [];
    let timeValues = [];

    function shuffleArray(array) {
      const shuffled = [...array];

      for (let i = shuffled.length - 1; i > 0; i--) {
        const randomIndex = Math.floor(Math.random() * (i + 1));
        [shuffled[i], shuffled[randomIndex]] = [shuffled[randomIndex], shuffled[i]];
      }

      return shuffled;
    }

    function loadOrCreateCard() {
      const savedCard = localStorage.getItem(`${GAME_ID}-phrases`);
      const savedMarks = localStorage.getItem(`${GAME_ID}-marks`);
      const savedTimes = localStorage.getItem(`${GAME_ID}-times`);

      if (savedCard) {
        cardPhrases = JSON.parse(savedCard);
      } else {
        cardPhrases = shuffleArray(phrases);
        localStorage.setItem(`${GAME_ID}-phrases`, JSON.stringify(cardPhrases));
      }

      markedSquares = savedMarks ? JSON.parse(savedMarks) : Array(9).fill(false);
      timeValues = savedTimes ? JSON.parse(savedTimes) : Array(9).fill("");
    }

    function saveProgress() {
      localStorage.setItem(`${GAME_ID}-marks`, JSON.stringify(markedSquares));
      localStorage.setItem(`${GAME_ID}-times`, JSON.stringify(timeValues));
    }

    function createCard() {
      card.innerHTML = "";

      cardPhrases.forEach((phrase, index) => {
        const square = document.createElement("div");
        square.classList.add("square");

        if (markedSquares[index]) {
          square.classList.add("marked");
        }

        square.innerHTML = `
          <div class="phrase">${phrase}</div>
          <div class="time-row">
            Time - 
            <input 
              class="time-input" 
              type="text" 
              placeholder="00:00"
              value="${timeValues[index]}"
            >
          </div>
        `;

        square.addEventListener("click", function () {
          markedSquares[index] = !markedSquares[index];
          square.classList.toggle("marked");
          saveProgress();
          checkBingo();
        });

        const input = square.querySelector(".time-input");

        input.addEventListener("click", function (event) {
          event.stopPropagation();
        });

        input.addEventListener("input", function () {
          timeValues[index] = input.value;
          saveProgress();
        });

        card.appendChild(square);
      });

      checkBingo();
    }

    function checkBingo() {
      const hasBingo = winningLines.some(line =>
        line.every(index => markedSquares[index])
      );

      if (hasBingo) {
        message.textContent = "🎉 BINGO! Type BINGO in Discord! 🎉";
      } else {
        message.textContent = "";
      }
    }

    function resetMarks() {
      markedSquares = Array(9).fill(false);
      timeValues = Array(9).fill("");
      saveProgress();
      createCard();
    }

    loadOrCreateCard();
    createCard();
  </script>
